# Mat:


    #!/bin/bash 
    
    lsof > lsof.log
    
    cat lsof.log | awk '{print $1}' | sort | uniq > item.list
    
    for item in `cat item.list`
    do
            echo -n "$item: "
            cat lsof.log | grep '^$item ' | wc -l
    done
